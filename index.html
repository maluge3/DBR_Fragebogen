1. Verbindung

Benötigte Tabelle in Supabase

create table if not exists questionnaire_responses (
  id bigint generated by default as identity primary key,
  session_id text not null,
  item1 int not null check (item1 between 0 and 100),
  item2 int not null check (item2 between 0 and 100),
  item3 int not null check (item3 between 0 and 100),
  item4 int not null check (item4 between 0 and 100),
  item5 int not null check (item5 between 0 and 100),
  created_at timestamptz default now()
);

alter table questionnaire_responses enable row level security;

create policy "anon read questionnaire_responses"
on questionnaire_responses for select
to anon using (true);

create policy "anon insert questionnaire_responses"
on questionnaire_responses for insert
to anon with check (true);
